# Building a Semantic Search Engine with Hugging Face Transformers and MongoDB Atlas Vector Search
**Author:** Arek Borucki  **Published:** February 10, 2026

## Overview
This tutorial demonstrates constructing a multilingual semantic search system using Hugging Face's transformer models and MongoDB Atlas Vector Search. The implementation enables meaning-based retrieval across different languages using the paraphrase-multilingual-MiniLM-L12-v2 model (supports 50+ languages), moving beyond traditional keyword matching to understand semantic intent.

## Key Concepts
- Semantic Search — maps text to numerical representations (embeddings) that capture meaning
- Embeddings — 384-dimension vector representations generated by language models
- Vector Database — MongoDB Atlas Vector Search for high-dimensional vector storage
- Multilingual Models — paraphrase-multilingual-MiniLM-L12-v2 supports 50+ languages
- Approximate Nearest Neighbor Search — finds semantically similar documents

## Code Examples

### Database Module
```python
import os
from pymongo import MongoClient
from pymongo.operations import SearchIndexModel
from dotenv import load_dotenv

load_dotenv()

class MongoManager:
    def __init__(self):
        self.client = MongoClient(os.getenv("MONGO_URI"))
        self.db = self.client[os.getenv("DB_NAME")]
        self.collection = self.db[os.getenv("COLLECTION_NAME")]

    def get_collection(self):
        return self.collection

    def create_vector_index(self):
        search_index_model = SearchIndexModel(
            definition={
                "fields": [
                    {
                        "type": "vector",
                        "path": "embedding",
                        "numDimensions": 384,
                        "similarity": "cosine"
                    }
                ]
            },
            name="vector_index",
            type="vectorSearch"
        )
        try:
            self.collection.create_search_index(model=search_index_model)
            print("Vector index 'vector_index' has been initialized.")
        except Exception as e:
            print(f"Index status: {e}")
```

### AI Model Module
```python
import os
from sentence_transformers import SentenceTransformer
from dotenv import load_dotenv

load_dotenv()

class AIModel:
    def __init__(self):
        model_name = os.getenv("MODEL_NAME")
        print(f"Loading Transformer model: {model_name}...")
        self.model = SentenceTransformer(model_name)
        print("Model ready.")

    def generate_embedding(self, text: str):
        if not text:
            return []
        return self.model.encode(text).tolist()
```

### FastAPI Search Endpoint
```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel, field_validator

app = FastAPI(title="Fashion Semantic Search")
db = MongoManager()
ai = AIModel()
collection = db.get_collection()

class SearchRequest(BaseModel):
    query: str
    limit: int = 5

    @field_validator('query')
    @classmethod
    def query_must_not_be_empty(cls, v):
        if not v or not v.strip():
            raise ValueError('Query cannot be empty')
        return v.strip()

@app.post("/search")
def search_products(payload: SearchRequest):
    query_vector = ai.generate_embedding(payload.query)

    pipeline = [
        {
            "$vectorSearch": {
                "index": "vector_index",
                "path": "embedding",
                "queryVector": query_vector,
                "numCandidates": 100,
                "limit": payload.limit
            }
        },
        {
            "$project": {
                "_id": 0,
                "name": 1,
                "category": 1,
                "subCategory": 1,
                "score": {"$meta": "vectorSearchScore"}
            }
        }
    ]

    results = list(collection.aggregate(pipeline))
    return {"query": payload.query, "results": results}
```
